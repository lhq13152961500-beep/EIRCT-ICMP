```js
// 直接内存转换，不使用 fs 写入读出
import ffmpeg from "fluent-ffmpeg";
import { PassThrough } from "stream";

async function convertM4aToPcmInMemory(m4aBuffer) {
  return new Promise((resolve) => {
    const pcmStream = new PassThrough();
    const chunks = [];
    pcmStream.on("data", (c) => chunks.push(c));
    pcmStream.on("end", () => resolve(Buffer.concat(chunks)));

    ffmpeg(Readable.from(m4aBuffer))
      .inputFormat("m4a")
      .outputFormat("s16le")
      .audioFrequency(16000)
      .audioChannels(1)
      .pipe(pcmStream);
  });
}
```

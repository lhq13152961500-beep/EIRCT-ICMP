### 2\. 完整后端代码（server\.js）

```javascript
const express = require('express');
const http = require('http');
const WebSocket = require('ws');
const axios = require('axios');

const app = express();
const server = http.createServer(app);
const wss = new WebSocket.Server({ server });

// 填入你的豆包 API_KEY
const API_KEY = "你的豆包API_KEY";
const BASE_URL = "https://ark.cn-beijing.volcesgate.com/api/v3";

wss.on('connection', (ws) => {
  console.log("终极语音客户端已连接");

  ws.on('message', async (data) => {
    try {
      const msg = JSON.parse(data.toString());

      if (msg.type === 'user_audio') {
        // ASR 语音转文字
        const asrRes = await axios.post(
          `${BASE_URL}/audio/speech-to-text`,
          { audio: msg.audioBase64 },
          { headers: { Authorization: `Bearer ${API_KEY}` } }
        );
        const query = asrRes.data.text?.trim();
        if (!query) return;

        // 豆包对话
        const chatRes = await axios.post(
          `${BASE_URL}/chat/completions`,
          {
            model: "doubao-lite-4k",
            messages: [{ role: "user", content: query }]
          },
          { headers: { Authorization: `Bearer ${API_KEY}` } }
        );
        const answer = chatRes.data.choices[0].message.content;

        // TTS 语音合成
        const ttsRes = await axios.post(
          `${BASE_URL}/audio/text-to-speech`,
          {
            text: answer,
            format: "mp3"
          },
          {
            headers: { Authorization: `Bearer ${API_KEY}` },
            responseType: 'json'
          }
        );

        // 回传结果
        ws.send(JSON.stringify({
          type: 'ai_voice',
          query,
          answer,
          voice: ttsRes.data.audio
        }));
      }
    } catch (e) {
      console.error("处理错误", e);
    }
  });

  ws.on('close', () => {
    console.log("客户端断开");
  });
});

const PORT = process.env.PORT || 3000;
server.listen(PORT, () => {
  console.log(`服务运行在端口 ${PORT}`);
});
```

### 2\. 完整 App 代码（App\.js）

```jsx
import { useState, useEffect, useRef } from 'react';
import { View, Text, StyleSheet, TouchableOpacity } from 'react-native';
import { Audio } from 'expo-av';
import WebSocket from 'ws';

export default function App() {
  const [enabled, setEnabled] = useState(true);
  const [status, setStatus] = useState('待机：说“你好豆包”唤醒');
  const [userText, setUserText] = useState('');
  const [aiText, setAiText] = useState('');

  const ws = useRef(null);
  const recording = useRef(null);
  const sound = useRef(null);
  const isListening = useRef(false);
  const isPlaying = useRef(false);

  // 唤醒词配置
  const WAKE_WORD = "你好豆包";

  // WebSocket 连接
  useEffect(() => {
    const socket = new WebSocket("wss://你的项目名.repl.co");
    ws.current = socket;

    socket.onmessage = async (e) => {
      const data = JSON.parse(e.data);
      if (data.type === 'ai_voice') {
        setUserText(data.query);
        setAiText(data.answer);
        await playVoice(data.voice);
      }
    };

    return () => socket.close();
  }, []);

  // 自动录音控制
  useEffect(() => {
    if (!enabled) {
      stopAll();
      return;
    }
    startLoopRecord();
  }, [enabled]);

  // 持续录音监听
  const startLoopRecord = async () => {
    try {
      await Audio.requestPermissionsAsync();
      await Audio.setAudioModeAsync({
        allowsRecordingIOS: true,
        playsInSilentModeIOS: true,
      });

      const { recording: rec } = await Audio.Recording.createAsync({
        sampleRate: 16000,
        numberOfChannels: 1,
        bitRate: 16000,
        format: 'wav',
      });
      recording.current = rec;

      // 每 800ms 分析音频
      setInterval(async () => {
        if (isPlaying.current) {
          // 检测到人声 → 打断 AI
          stopVoice();
          isPlaying.current = false;
          setStatus("已打断，聆听中…");
          isListening.current = true;
        }
        await analyzeChunk();
      }, 800);
    } catch (err) {
      console.error(err);
    }
  };

  // 音频分析并上传
  const analyzeChunk = async () => {
    if (!recording.current) return;
    const uri = recording.current.getURI();
    if (!uri) return;

    const res = await fetch(uri);
    const blob = await res.blob();

    const reader = new FileReader();
    reader.readAsDataURL(blob);
    reader.onloadend = () => {
      const base64 = reader.result?.split(',')[1];
      if (!base64) return;

      ws.current?.send(JSON.stringify({
        type: 'user_audio',
        audioBase64: base64,
        isWakeCheck: !isListening.current
      }));
    };
  };

  // 播放 AI 回复语音
  const playVoice = async (base64) => {
    stopVoice();
    isPlaying.current = true;
    setStatus("AI 回答中…");

    const { sound: s } = await Audio.Sound.createAsync({
      uri: `data:audio/mp3;base64,${base64}`
    });
    sound.current = s;
    await s.playAsync();

    s.setOnPlaybackStatusUpdate((status) => {
      if (status.didJustFinish) {
        isPlaying.current = false;
        isListening.current = false;
        setStatus('待机：说“你好豆包”唤醒');
      }
    });
  };

  // 停止播放
  const stopVoice = async () => {
    try {
      await sound.current?.stopAsync();
    } catch {}
  };

  // 停止所有
  const stopAll = async () => {
    stopVoice();
    await recording.current?.stopAndUnloadAsync();
    setStatus("已关闭");
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>豆包终极语音助手</Text>
      <Text style={styles.status}>{status}</Text>

      <TouchableOpacity
        style={styles.toggleBtn}
        onPress={() => setEnabled(!enabled)}>
        <Text style={styles.toggleText}>
          {enabled ? '关闭助手' : '开启助手'}
        </Text>
      </TouchableOpacity>

      <Text style={styles.userText}>你：{userText}</Text>
      <Text style={styles.aiText}>豆包：{aiText}</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 20,
    justifyContent: 'center',
    backgroundColor: '#f5f5f5',
  },
  title: {
    fontSize: 22,
    fontWeight: 'bold',
    textAlign: 'center',
    marginBottom: 10,
  },
  status: {
    fontSize: 14,
    color: '#666',
    textAlign: 'center',
    marginBottom: 30,
  },
  toggleBtn: {
    backgroundColor: '#007AFF',
    padding: 15,
    borderRadius: 12,
    alignItems: 'center',
    marginBottom: 40,
  },
  toggleText: {
    color: '#fff',
    fontSize: 16,
    fontWeight: 'bold',
  },
  userText: {
    fontSize: 16,
    marginVertical: 4,
  },
  aiText: {
    fontSize: 16,
    color: '#007AFF',
    marginVertical: 4,
  },
});
```
